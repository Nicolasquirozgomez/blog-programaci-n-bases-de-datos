Introducción

Ejemplos de Condicionales y Bucles en MySQL

Condicional IF THEN ELSE

El condicional IF THEN ELSE en MySQL se utiliza para ejecutar una acción cuando se cumple una condición específica, y una acción diferente si no se cumple.

-- Ejemplo 1: Comprobación de saldo en una cuenta
DELIMITER $$

CREATE PROCEDURE VerificarSaldo()
BEGIN
    DECLARE saldo INT DEFAULT 100;
    
    IF saldo > 0 THEN
        SELECT 'El saldo es suficiente para realizar la compra';
    ELSE
        SELECT 'No hay suficiente saldo';
    END IF;
END $$

DELIMITER ;

Este ejemplo verifica si el saldo de una cuenta es mayor a 0. Si es así, permite realizar una compra, de lo contrario, muestra un mensaje indicando que no hay suficiente saldo.

-- Ejemplo 2: Verificar edad para acceder a un sistema
DELIMITER $$

CREATE PROCEDURE VerificarEdad(edad INT)
BEGIN
    IF edad >= 18 THEN
        SELECT 'Acceso permitido';
    ELSE
        SELECT 'Acceso denegado';
    END IF;
END $$

DELIMITER ;

Aquí, verificamos si la edad de una persona es mayor o igual a 18. Si cumple con este requisito, se le permite el acceso, de lo contrario, se le niega el acceso.

-- Ejemplo 3: Determinar si un número es par o impar
DELIMITER $$

CREATE PROCEDURE VerificarParImpar(numero INT)
BEGIN
    IF numero % 2 = 0 THEN
        SELECT 'El número es par';
    ELSE
        SELECT 'El número es impar';
    END IF;
END $$

DELIMITER ;

Este ejemplo muestra cómo verificar si un número dado es par o impar utilizando el operador módulo (%).

-- Ejemplo 4: Validar si un usuario está activo
DELIMITER $$

CREATE PROCEDURE ValidarUsuario(usuario INT)
BEGIN
    IF usuario = 1 THEN
        SELECT 'El usuario está activo';
    ELSE
        SELECT 'El usuario no está activo';
    END IF;
END $$

DELIMITER ;

En este ejemplo se valida si un usuario está activo en el sistema, según su identificador (1 para activo, otros valores para no activo).

-- Ejemplo 5: Asignar un mensaje según el día de la semana
DELIMITER $$

CREATE PROCEDURE MensajeDiaSemana(dia INT)
BEGIN
    IF dia = 1 THEN
        SELECT 'Hoy es lunes';
    ELSEIF dia = 2 THEN
        SELECT 'Hoy es martes';
    ELSE
        SELECT 'Hoy es otro día';
    END IF;
END $$

DELIMITER ;

Este ejemplo muestra cómo usar una serie de IF ELSEIF para asignar un mensaje diferente dependiendo del día de la semana.

Bucles REPEAT

El bucle REPEAT ejecuta un bloque de código repetidamente hasta que se cumpla una condición.

-- Ejemplo 1: Contador con REPEAT
DELIMITER $$

CREATE PROCEDURE ContadorRepetido()
BEGIN
    DECLARE contador INT DEFAULT 1;
    
    REPEAT
        SELECT contador;
        SET contador = contador + 1;
    UNTIL contador > 10
    END REPEAT;
END $$

DELIMITER ;

Este ejemplo cuenta de 1 a 10 utilizando el bucle REPEAT.

-- Ejemplo 2: Acumulación de total con REPEAT
DELIMITER $$

CREATE PROCEDURE AcumulacionTotal()
BEGIN
    DECLARE total INT DEFAULT 0;
    DECLARE cantidad INT DEFAULT 10;
    
    REPEAT
        SET total = total + cantidad;
        SET cantidad = cantidad - 1;
    UNTIL cantidad = 0
    END REPEAT;
    
    SELECT total;
END $$

DELIMITER ;

Aquí usamos REPEAT para acumular el valor de cantidad en total, decrementando cantidad en cada iteración.

-- Ejemplo 3: Bucle REPEAT con condición al final
DELIMITER $$

CREATE PROCEDURE BucleHasta10()
BEGIN
    DECLARE contador INT DEFAULT 1;
    
    REPEAT
        SELECT contador;
        SET contador = contador + 1;
    UNTIL contador > 10
    END REPEAT;
END $$

DELIMITER ;

Este ejemplo muestra cómo imprimir números del 1 al 10 utilizando el bucle REPEAT, donde la condición se verifica después de cada iteración.

-- Ejemplo 4: Repetir hasta encontrar un valor específico
DELIMITER $$

CREATE PROCEDURE BuscarValor()
BEGIN
    DECLARE encontrado BOOLEAN DEFAULT FALSE;
    DECLARE valor INT DEFAULT 3;
    
    REPEAT
        IF valor = 3 THEN
            SET encontrado = TRUE;
        ELSE
            SET valor = valor + 1;
        END IF;
    UNTIL encontrado = TRUE
    END REPEAT;
    
    SELECT 'Valor encontrado: ', valor;
END $$

DELIMITER ;

Este ejemplo demuestra cómo usar REPEAT para seguir buscando un valor específico hasta encontrarlo.

-- Ejemplo 5: Contar números positivos con REPEAT
DELIMITER $$

CREATE PROCEDURE ContarPositivos()
BEGIN
    DECLARE numero INT DEFAULT -5;
    
    REPEAT
        SET numero = numero + 1;
        SELECT numero;
    UNTIL numero >= 5
    END REPEAT;
END $$

DELIMITER ;

Este ejemplo muestra cómo contar de -5 hasta 5 utilizando REPEAT.

Bucles LOOP

El bucle LOOP ejecuta un bloque de código de forma indefinida, hasta que se alcanza una condición de salida utilizando LEAVE.

-- Ejemplo 1: Bucle LOOP para contar hasta 10
DELIMITER $$

CREATE PROCEDURE ContarHasta10Loop()
BEGIN
    DECLARE contador INT DEFAULT 1;
    
    LOOP
        SELECT contador;
        SET contador = contador + 1;
        IF contador > 10 THEN
            LEAVE;
        END IF;
    END LOOP;
END $$

DELIMITER ;

Este ejemplo utiliza el bucle LOOP para contar del 1 al 10 y luego salir del bucle cuando el contador alcanza 10.

-- Ejemplo 2: Bucle LOOP con condiciones de búsqueda
DELIMITER $$

CREATE PROCEDURE BuscarNumero()
BEGIN
    DECLARE numero INT DEFAULT 1;
    
    LOOP
        IF numero = 5 THEN
            SELECT 'Número encontrado: ', numero;
            LEAVE;
        END IF;
        SET numero = numero + 1;
    END LOOP;
END $$

DELIMITER ;

En este ejemplo, el bucle LOOP busca el número 5 y sale del bucle cuando lo encuentra.

-- Ejemplo 3: Bucle LOOP para sumar números
DELIMITER $$

CREATE PROCEDURE SumarNumeros()
BEGIN
    DECLARE suma INT DEFAULT 0;
    DECLARE numero INT DEFAULT 1;
    
    LOOP
        SET suma = suma + numero;
        SET numero = numero + 1;
        IF numero > 5 THEN
            LEAVE;
        END IF;
    END LOOP;
    
    SELECT suma;
END $$

DELIMITER ;

Este ejemplo muestra cómo usar un bucle LOOP para sumar los números del 1 al 5.

-- Ejemplo 4: Bucle LOOP con condición de parada
DELIMITER $$

CREATE PROCEDURE SumarHastaFin()
BEGIN
    DECLARE suma INT DEFAULT 0;
    DECLARE fin INT DEFAULT 10;
    
    LOOP
        SET suma = suma + fin;
        SET fin = fin - 1;
        IF fin = 0 THEN
            LEAVE;
        END IF;
    END LOOP;
    
    SELECT suma;
END $$

DELIMITER ;

Este ejemplo muestra cómo restar valores de una variable fin hasta llegar a 0, acumulando la suma en cada paso.

-- Ejemplo 5: Generación de secuencia con LOOP
DELIMITER $$

CREATE PROCEDURE SecuenciaLoop()
BEGIN
    DECLARE i INT DEFAULT 1;
    
    LOOP
        SELECT i;
        SET i = i + 1;
        IF i > 5 THEN
            LEAVE;
        END IF;
    END LOOP;
END $$

DELIMITER ;

Este ejemplo genera una secuencia de números del 1 al 5 utilizando un bucle LOOP.

Bucles WHILE

El bucle WHILE ejecuta un bloque de código mientras se cumpla una condición. Es útil cuando se desconoce cuántas veces se debe iterar.

-- Ejemplo 1: Imprimir números del 1 al 5 con WHILE
DELIMITER $$

CREATE PROCEDURE ImprimirHasta5()
BEGIN
    DECLARE contador INT DEFAULT 1;
    
    WHILE contador <= 5 DO
        SELECT contador;
        SET contador = contador + 1;
    END WHILE;
END $$

DELIMITER ;

Este ejemplo muestra cómo usar un bucle WHILE para imprimir los números del 1 al 5.

-- Ejemplo 2: Sumar números del 1 al 10 con WHILE
DELIMITER $$

CREATE PROCEDURE SumarHasta10()
BEGIN
    DECLARE suma INT DEFAULT 0;
    DECLARE contador INT DEFAULT 1;
    
    WHILE contador <= 10 DO
        SET suma = suma + contador;
        SET contador = contador + 1;
    END WHILE;
    
    SELECT suma;
END $$

DELIMITER ;

Este ejemplo suma los números del 1 al 10 utilizando un bucle WHILE y luego muestra el resultado.

-- Ejemplo 3: Encontrar el primer número mayor a 10 con WHILE
DELIMITER $$

CREATE PROCEDURE EncontrarMayor()
BEGIN
    DECLARE numero INT DEFAULT 1;
    
    WHILE numero <= 10 DO
        SET numero = numero + 1;
    END WHILE;
    
    SELECT 'El primer número mayor a 10 es: ', numero;
END $$

DELIMITER ;

Este ejemplo muestra cómo encontrar el primer número mayor a 10 utilizando el bucle WHILE.

-- Ejemplo 4: Decrementar hasta llegar a 0 con WHILE
DELIMITER $$

CREATE PROCEDURE ContarRegresivo()
BEGIN
    DECLARE contador INT DEFAULT 5;
    
    WHILE contador >= 0 DO
        SELECT contador;
        SET contador = contador - 1;
    END WHILE;
END $$

DELIMITER ;

Este ejemplo muestra cómo realizar un conteo regresivo desde 5 hasta 0 utilizando el bucle WHILE.

-- Ejemplo 5: Contar números positivos con WHILE
DELIMITER $$

CREATE PROCEDURE ContarPositivos()
BEGIN
    DECLARE numero INT DEFAULT -3;
    
    WHILE numero < 3 DO
        SET numero = numero + 1;
        SELECT numero;
    END WHILE;
END $$

DELIMITER ;

Este ejemplo utiliza un bucle WHILE para contar de -3 hasta 3 e imprimir cada valor.
